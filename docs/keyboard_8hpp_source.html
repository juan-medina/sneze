270struct key_modifier {
272 code key = key::unknown; // NOLINT(misc-non-private-member-variables-in-classes)
274 mod modifier = modifier::none; // NOLINT(misc-non-private-member-variables-in-classes)
275
281 constexpr auto operator==(const key_modifier &other) const {
282 if(other.modifier == keyboard::modifier::none) {
283 return key == other.key && modifier == keyboard::modifier::none;
284 }
285 return (key == other.key) && ((modifier & other.modifier) != 0);
286 }
287
293 constexpr auto operator==(const code &other) const {
294 return key == other && modifier == keyboard::modifier::none;
295 }
296
301 [[nodiscard]] auto string() const {
302 auto key_str = key::string(key);
303 if(auto mod_str = modifier ::string(modifier); !mod_str.empty()) {
304 return mod_str + "+" + key_str;
305 }
306 return key_str;
307 }
308};
