113 template<typename... Args>
114 static void
115 log(const level level, std::source_location source, fmt::format_string<Args...> fmt, Args &&...args) {
116  auto location = spdlog::source_loc{source.file_name(), static_cast<int>(source.line()), source.function_name()};
117  spdlog::log(location, static_cast<spdlog::level::level_enum>(level), fmt, std::forward<Args>(args)...);
118 }
