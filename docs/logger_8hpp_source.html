96enum class level : int {
98 trace = spdlog::level::trace,
100 debug = spdlog::level::debug,
102 info = spdlog::level::info,
104 warning = spdlog::level::warn,
106 error = spdlog::level::err,
108 critical = spdlog::level::critical,
110 off = spdlog::level::off
111};

113template<typename... Args>
114static void
115log(const level level, std::source_location source, fmt::format_string<Args...> fmt, Args &&...args) {
116 auto location = spdlog::source_loc{source.file_name(), static_cast<int>(source.line()), source.function_name()};
117 spdlog::log(location, static_cast<spdlog::level::level_enum>(level), fmt, std::forward<Args>(args)...);
118}
