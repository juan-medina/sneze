83template<typename... Args>
84static void
85log(const level::log_level level, std::source_location source, fmt::format_string<Args...> fmt, Args &&...args) {
86 auto location = spdlog::source_loc{source.file_name(), static_cast<int>(source.line()), source.function_name()};
87 spdlog::log(location, static_cast<spdlog::level::level_enum>(level), fmt, std::forward<Args>(args)...);
88}
