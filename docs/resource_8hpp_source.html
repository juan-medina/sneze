135 [[nodiscard]] auto load(const std::string &uri, Args... args) -> result<> {
136 if(auto it_resource = resources_.find(uri); it_resource != resources_.end()) {
137 it_resource->second.count++;
138 logger::trace("request to load resource<{}>: {}, increase count to: {}",
139 type_name<Type>(),
140 uri,
141 it_resource->second.count);
142 } else {
143 auto new_resource = std::make_shared<Type>(render_);
144 if(auto err = new_resource->init(uri, args...).ko(); err) {
145 new_resource.reset();
146 logger::error("fail to load resource<{}>: {}", type_name<Type>(), uri);
147 return error("Fail to load resource.", *err);
148 }
149 logger::trace("new resource<{}> loaded: {}, set count to: 1", type_name<Type>(), uri);
150 resources_.insert({uri, resource_entry<Type, Args...>{new_resource, 1}});
151 }
152 return true;
153 }
154
161 [[nodiscard]] auto unload(const std::string &uri) -> result<> {
162 if(auto it_resource = resources_.find(uri); it_resource != resources_.end()) {
163 it_resource->second.count--;
164 logger::trace("request to unload resource<{}>: {}, decrease count to: {}",
165 type_name<Type>(),
166 uri,
167 it_resource->second.count);
168 if(it_resource->second.count == 0) {
169 logger::trace("resource<{}> unloaded: {} since count is 0", type_name<Type>(), uri);
170 resources_.erase(it_resource);
171 }
172 return true;
173 }
174 logger::error("fail to unload resource<{}>: {}", type_name<Type>(), uri);
175 return error("Fail to unload resource.");
176 }
177
184 [[nodiscard]] auto get(const std::string &uri) -> result<std::shared_ptr<Type>, error> {
185 if(auto it_resource = resources_.find(uri); it_resource != resources_.end()) {
186 return it_resource->second.data;
187 }
188 logger::error("fail to get a resource<{}> not loaded: {}", type_name<Type>(), uri);
189 return error("Fail to get resource.");
190 }
191
195 void clear() {
196 resources_.clear();
197 }
198
199private:
201 std::unordered_map<std::string, resource_entry<Type, Args...>> resources_ = {};
203 render *render_{nullptr};
204};
