64 [[nodiscard]] static inline auto rgba(const component &red_component,
65 const component &green_component,
66 const component &blue_component,
67 const component &alpha_component) -> color {
68 return {red_component, green_component, blue_component, alpha_component};
69 }
70
79 [[nodiscard]] static inline auto
80 rgb(const component &red_component, const component &green_component, const component &blue_component) {
81 return rgba(red_component, green_component, blue_component, opacity::opaque);
82 }
83
89 [[nodiscard]] inline auto alpha(component alpha) const -> color {
90 return rgba(r, g, b, alpha);
91 }
92
98 [[nodiscard]] inline auto alpha(float alpha) const -> color {
99 return rgba(r, g, b, static_cast<component>(alpha * static_cast<component>(opacity::opaque)));
100 }
101
107 [[nodiscard]] inline auto alpha(opacity alpha) const -> color {
108 return rgba(r, g, b, static_cast<component>(alpha));
109 }
110
116 void blend(const color &other, float factor) {
117 r = static_cast<component>(static_cast<float>(r) * (1.0F - factor) + static_cast<float>(other.r) * factor);
118 g = static_cast<component>(static_cast<float>(g) * (1.0F - factor) + static_cast<float>(other.g) * factor);
119 b = static_cast<component>(static_cast<float>(b) * (1.0F - factor) + static_cast<float>(other.b) * factor);
120 a = static_cast<component>(static_cast<float>(a) * (1.0F - factor) + static_cast<float>(other.a) * factor);
121 }
